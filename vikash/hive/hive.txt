🔹 STEP 1: START HIVE

Open terminal and type:

hive
🔹 STEP 2: CREATE DATABASE

👉 A database is like a folder

CREATE DATABASE flight_db;
USE flight_db;
🔹 STEP 3: CREATE TABLE (FLIGHT DATA)

👉 This stores flight details

CREATE TABLE flights (
year INT,
month INT,
day INT,
carrier STRING,
flight_num INT,
origin STRING,
dest STRING,
dep_delay FLOAT,
arr_delay FLOAT
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ','
STORED AS TEXTFILE;
🔹 STEP 4: DROP TABLE (if needed)
DROP TABLE flights;
🔹 STEP 5: ALTER TABLE

👉 Add new column

ALTER TABLE flights ADD COLUMNS (distance INT);
🔹 STEP 6: CREATE EXTERNAL TABLE

👉 Data stays even if table is deleted

CREATE EXTERNAL TABLE flights_ext (
year INT,
month INT,
day INT,
carrier STRING,
flight_num INT,
origin STRING,
dest STRING,
dep_delay FLOAT,
arr_delay FLOAT
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ','
LOCATION '/user/hadoop/flight_data/';
🔹 STEP 7: LOAD DATA

👉 Load CSV file

LOAD DATA LOCAL INPATH '/home/hadoop/flights.csv'
INTO TABLE flights;
🔹 STEP 8: INSERT DATA
INSERT INTO flights VALUES (2008,1,1,'AA',101,'NYC','LAX',5,10);
🔹 STEP 9: JOIN TABLES

👉 Create another table

CREATE TABLE airlines (
carrier STRING,
name STRING
);

INSERT INTO airlines VALUES ('AA','American Airlines');

👉 Join query

SELECT f.carrier, a.name, f.dep_delay
FROM flights f
JOIN airlines a
ON f.carrier = a.carrier;
🔹 STEP 10: CREATE INDEX
CREATE INDEX idx_flight
ON TABLE flights (carrier)
AS 'COMPACT'
WITH DEFERRED REBUILD;

ALTER INDEX idx_flight ON flights REBUILD;
🔹 STEP 11: FIND AVERAGE DEPARTURE DELAY (MAIN QUESTION)
SELECT day, AVG(dep_delay) AS avg_delay
FROM flights
WHERE year = 2008
GROUP BY day;
