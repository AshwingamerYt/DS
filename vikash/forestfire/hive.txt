CREATE TABLE forestfires (
X INT,
Y INT,
month STRING,
day STRING,
FFMC FLOAT,
DMC FLOAT,
DC FLOAT,
ISI FLOAT,
temp FLOAT,
RH FLOAT,
wind FLOAT,
rain FLOAT,
area FLOAT
)
ROW FORMAT SERDE 'org.apache.hadoop.hive.serde2.OpenCSVSerde'
WITH SERDEPROPERTIES (
  "separatorChar" = ",",
  "quoteChar" = "\""
)
STORED AS TEXTFILE
TBLPROPERTIES ("skip.header.line.count"="1");
LOAD DATA INPATH '/input/forestfires.csv'
INTO TABLE forestfires;
SELECT month, SUM(area)
FROM forestfires
GROUP BY month;
SELECT month, COUNT(*)
FROM forestfires
GROUP BY month;
SELECT month, AVG(temp) AS avg_temp
FROM forestfires
GROUP BY month
ORDER BY avg_temp DESC;
